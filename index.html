# Below contains basic text processing functions
def preprocess(s, lowercase=True, strip_punctuation=True):
    """
    Input:
        string s
        boolean lowercase
        boolean strip_punctuation
    Return:
        list of strings after preprocessing
    """
    punctuation = '.,?<>:;"\'!%[]_`*-'
    if isinstance(s, str):
        s = tokenize(s)
    if lowercase:
        s = [t.lower() for t in s]
    if strip_punctuation:
        s = [t.strip(punctuation) for t in s]
    s = [t for t in s if t !='']
        
    return s

def tokenize(text):
    return text.split()

def token_frequency(text, relative=False):
    """
    Input: 
        text - list of strings
    Return:
        freqs - a dictionary contains the token_frequency
    """
    freqs = {}
    for token in text:
        if token in freqs:
            freqs[token] += 1
        else:
            freqs[token] = 1
    if relative:
        total = len(text)
        for token in freqs:
            freqs[token] = freqs[token]/total
    return freqs

def document_frequency(tfs_list):
    """
    Input: 
        tfs_list - a list of token frequencies
    Return:
        df - a dictionary contains the document frequency
    """
    df = {}
    for tf in tfs_list:
        for token in tf:
            if token in df:
                df[token] += 1
            else:
                df[token] = 1
    return df

def cartesian_distance(v1, v2):
    """
    Input: 
        v1, v2 - two dictionaries with feature & frequency pairs
    Return:
        cartesian_distance between v1 and v2
    """
    dist = 0
    for feature in set(list(v1.keys())+list(v2.keys())):
        if feature in v1 and feature in v2:
            dist += (v1[feature]-v2[feature])**2
        elif feature in v1 and feature not in v2:
            dist += (v1[feature]-0)**2
        else:
            dist += (v2[feature]-0)**2
    return dist**0.5

def token_group_frequency(tf, featureset_list):
    """
    Input: 
        tf - a dictionary of token frequency
        featureset_list - a list of list of features as a group
    Return:
        vec - a list contains the token group frequency
    """
    vec = []
    
    for featureset in featureset_list:
        f = 0
        for feature in featureset:
            if feature in tf:
                f += tf[feature]
        vec.append(f)
    return vec

def chunk_corpus(text, chunksize):
    """
    Input: 
        text - list of strings
        chunksize - int
    Return:
        chunks - a list of strings where each string is a chunk of chunksize
    """
    chunks = []
    for start in range(0, len(text)-chunksize+1, chunksize):
        chunk = text[start:start+chunksize]
        chunks.append(chunk)
    return chunks

for fp in file_paths:
    tokens = preprocess(open(fp, 'r').read())
    novel_token_freqs.append(token_frequency(tokens, relative=True)) #Get the token frequency of novels

df = document_frequency(novel_token_freqs) #Get the document frequency
corpus_tfidf_features = [] 

# calculate the tfidf scores in each novel
for novel in novel_token_freqs:    
    tfidf = {}
    for token in novel:
        token_tfidf = novel[token] * math.log((len(file_paths)/df[token]))
        tfidf[token] = token_tfidf
    # sorted the tfidf scores to get the top 100 tfidf words    
    tfidf_sorted = sorted(tfidf.items(), key = lambda x: x[1], reverse=True)[:100]
    # eliminate words with a document frequency <= 2
    tfidf_subset = [w for w,tfidf in tfidf_sorted if df[w]>2]
    # add to the feature set as a feature
    corpus_tfidf_features.append(tfidf_subset)

# get the vector for each chunk based on the top tfidf word group feature set 
corpus_tfidf_vectors = []
for c in corpus:
    tf = token_frequency(c)
    tf_group = token_group_frequency(tf, corpus_tfidf_features)
    corpus_tfidf_vectors.append(tf_group)
